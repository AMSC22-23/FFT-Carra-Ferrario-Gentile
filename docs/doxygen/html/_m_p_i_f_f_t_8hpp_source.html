90 template <typename FloatingType>
91 void MPIFFT<FloatingType>::fft(CTensor_1D &global_tensor, FFTDirection fftDirection) const
92 {
93
94 // Utility
95 using Complex = std::complex<FloatingType>;
96
97 // MPI Infos
98 int rank, size;
99 MPI_Datatype mpi_datatype = std::is_same<FloatingType, double>::value ? MPI_C_DOUBLE_COMPLEX : MPI_C_FLOAT_COMPLEX;
100 MPI_Comm_rank(MPI_COMM_WORLD, &rank);
101 MPI_Comm_size(MPI_COMM_WORLD, &size);
102
103 // Tensor infos
104 int n = global_tensor.size();
105 int log2n = std::log2(n);
106 double log2p = std::log2(size);
107
108 // Assertions
109 assert(!(n & (n - 1)) && "FFT length must be a power of 2.");
110 //assert((size <= n / 2 && std::ceil(log2p) == std::floor(log2p)) && "Process number must be a power of two and less or equal than tensor.size()/2).");
111 ASSERT(size, <= , n/2);
112 ASSERT(std::ceil(log2p), ==, std::floor(log2p));
113 // -------------------------------------
114
115 // Length of the sub-vector related to each process
116 int local_tensor_size = global_tensor.size() / size;
117 // Global index of the sub-vector
118 int offset = rank * local_tensor_size;
119
120 // Every process runs a bit reversal permutation towards
121 // a sub-vector, still by working on its copy of the global
122 // vector.
123
124 int rev;
125 for (int i = 0; i < local_tensor_size; i++)
126 {
127 rev = FFTUtils::reverseBits(offset + i, log2n);
128
129 // Avoid to swap elements if the reverse
130 // has already been processed by the process ( rev > (offset+i) ),
131 // or if it's placed in a sub vector
132 // of another process competence ( rev < offset )
133
134 if (rev > (offset + i) || rev < offset)
135 {
136
137 // If the elements to swap are in the same range
138 // of the process, swap them.
139
140 if ((rev / local_tensor_size) == rank)
141 {
142 std::swap(global_tensor[offset + i], global_tensor[rev]);
143 }
144 else
145 {
146
147 // Else, there is no need to copy the current position
148 // in the reversed position. This is because it will be
149 // discarded at MPI_Gather as it's not this processor competence
150 // to handle it.
151
152 global_tensor[offset + i] = global_tensor[rev];
153 }
154 }
155 }
156
157 // Gather the permuted sub-vectors of each processor competence into process 0.
158 MPI_Gather(global_tensor.data() + offset, local_tensor_size, mpi_datatype, global_tensor.data(), local_tensor_size, mpi_datatype, 0, MPI_COMM_WORLD);
159
160 // Conjugate
161 if (rank == 0)
162 {
163 if (fftDirection == FFT_INVERSE)
164 {
165 FFTUtils::conjugate(global_tensor);
166 }
167 }
168
169 // Local section of the vector
170 CTensor_1D local_tensor(local_tensor_size);
171
172
173
174
175 // Scatter the permutated (and conjugated) global tensor on all processes.
176 MPI_Scatter(global_tensor.data(), local_tensor_size, mpi_datatype, local_tensor.data(), local_tensor_size, mpi_datatype, 0, MPI_COMM_WORLD);
177
178 // Run fft on local tensors ( subtrees )
179 MPIFFT::_generalized_cooleytukey_butterfly(local_tensor, 1);
180
181 // Reconstruct the partially computed vector
182 MPI_Gather(local_tensor.data(), local_tensor_size, mpi_datatype, global_tensor.data(), local_tensor_size, mpi_datatype, 0, MPI_COMM_WORLD);
183
184 // Sequentially compute the remaining steps
185 if (rank == 0)
186 {
187
188 // Now there remains log2(num of processes) steps to do
189 // If there is only one process, butterfly has already been completed
190 if (log2p > 0)
191 MPIFFT::_generalized_cooleytukey_butterfly(global_tensor, log2n - log2p + 1);
192
193 // Re-conjugate and scale if inverse
194 if (fftDirection == FFT_INVERSE)
195 {
196 FFTUtils::conjugate(global_tensor);
197 global_tensor = global_tensor * Complex(1.0 / n, 0);
198 }
199 }
200
201 // The global vector is ready on process 0.
202 }

211 template <typename FloatingType>
212 void MPIFFT<FloatingType>::_generalized_cooleytukey_butterfly(CTensor_1D &input_output, const int starting_depth) const
213 {
214 using Complex = std::complex<FloatingType>;
215 int n = input_output.size();
216 assert(!(n & (n - 1)) && "FFT length must be a power of 2.");
217 int log2n = std::log2(n);
218 assert((starting_depth >= 1 && starting_depth <= log2n) && "Starting depth has to be in (1...log2(n)).");
219
220 Complex w, wm, t, u;
221 int m, m2;
222 // Cooley-Tukey iterative FFT
223 for (int s = starting_depth; s <= log2n; ++s)
224 {
225 m = 1 << s; // 2 power s
226 m2 = m >> 1; // m2 = m/2 -1
227 wm = exp(Complex(0, -2 * M_PI / m)); // w_m = e^(-2*pi/m)
228
229 for (int k = 0; k < n; k += m)
230 {
231 w = Complex(1, 0);
232 for (int j = 0; j < m2; ++j)
233 {
234 t = w * input_output[k + j + m2];
235 u = input_output[k + j];
236
237 input_output[k + j] = u + t;
238 input_output[k + j + m2] = u - t;
239
240 w *= wm;
241 }
242 }
243 }
244 };
