211  template <typename FloatingType>
212  void MPIFFT<FloatingType>::_generalized_cooleytukey_butterfly(CTensor_1D &input_output, const int starting_depth) const
213  {
214  using Complex = std::complex<FloatingType>;
215  int n = input_output.size();
216  assert(!(n & (n - 1)) && "FFT length must be a power of 2.");
217  int log2n = std::log2(n);
218  assert((starting_depth >= 1 && starting_depth <= log2n) && "Starting depth has to be in (1...log2(n)).");
219 
220  Complex w, wm, t, u;
221  int m, m2;
222  // Cooley-Tukey iterative FFT
223  for (int s = starting_depth; s <= log2n; ++s)
224  {
225  m = 1 << s; // 2 power s
226  m2 = m >> 1; // m2 = m/2 -1
227  wm = exp(Complex(0, -2 * M_PI / m)); // w_m = e^(-2*pi/m)
228 
229  for (int k = 0; k < n; k += m)
230  {
231  w = Complex(1, 0);
232  for (int j = 0; j < m2; ++j)
233  {
234  t = w * input_output[k + j + m2];
235  u = input_output[k + j];
236 
237  input_output[k + j] = u + t;
238  input_output[k + j + m2] = u - t;
239 
240  w *= wm;
241  }
242  }
243  }
244  };
