126 template <typename FloatingType>
127 void MPI_OMP_FFT_2D<FloatingType>::fft(CTensor_2D &global_tensor, FFTDirection fftDirection) const
128 {
129
130 // initialize mpi data
131 int rank, size;
132 MPI_Comm_rank(MPI_COMM_WORLD, &rank);
133 MPI_Comm_size(MPI_COMM_WORLD, &size);
134 MPI_Datatype mpi_datatype = std::is_same<FloatingType, double>::value ? MPI_C_DOUBLE_COMPLEX : MPI_C_FLOAT_COMPLEX;
135
136 assert(!(size & (size - 1)) && "MPI SIZE must be a power of 2.");
137
138 unsigned int x_dimension_size, y_dimension_size;
139 void* tensor_data;
140
141 OmpFFT omp_1D_solver;
142 CTensor_1D axis_input;
143
144 if(rank==0){
145 const auto& dimensions = global_tensor.dimensions();
146 x_dimension_size = dimensions[0];
147 y_dimension_size = dimensions[1];
148
149 //input dimensions check
150 for(int i=0; i<2; i++){
151 assert(!(dimensions[i] & (dimensions[i] - 1)) && "All dimensions of the input must be a power of 2.");
152 }
153
154 tensor_data = global_tensor.data();
155 }
156
157 //##########################################################################
158 // first phase
159 //##########################################################################
160
161 //--------------------------------------------------------------------------
162 // distribution
163 //--------------------------------------------------------------------------
164
165 //broadcast x and y tensor's dimension
166 MPI_Bcast(&x_dimension_size, 1, MPI_UNSIGNED, 0, MPI_COMM_WORLD);
167 MPI_Bcast(&y_dimension_size, 1, MPI_UNSIGNED, 0, MPI_COMM_WORLD);
168
169 //calculate block size
170 const unsigned int x_block_size = x_dimension_size/size;
171 const unsigned int y_block_size = y_dimension_size/size;
172
173 // Create the block disposition of the first phase of the transformation.
174 // Each element of block_distribution represents the block owner's rank.
175 Eigen::MatrixXi block_distribution;
176 block_distribution.resize(size, size);
177
178 // col-major displacement of blocks
179 for(int i=0; i<size; i++){
180 for(int j=0; j<size; j++){
181 block_distribution(i,j)=j;
182 }
183 }
184
185 // dimension of the local tensor
186 const unsigned int x_local_dimension_size=x_dimension_size;
187 const unsigned int y_local_dimension_size=y_dimension_size/size;
188 const unsigned int local_tensor_elements_number = x_local_dimension_size*y_local_dimension_size;
189
190 //local tensor
191 CTensor_2D local_tensor(x_local_dimension_size, y_local_dimension_size);
192 void* local_tensor_data = local_tensor.data();
193
194 MPI_Scatter(tensor_data, local_tensor_elements_number, mpi_datatype, local_tensor_data, local_tensor_elements_number, mpi_datatype,0, MPI_COMM_WORLD);
195
196
197 //--------------------------------------------------------------------------
198 // transformation
199 //--------------------------------------------------------------------------
200 for(Eigen::Index line=0; line< y_local_dimension_size; line++){
201 axis_input = local_tensor.chip(line,1);
202 omp_1D_solver.fft(axis_input, fftDirection);
203 local_tensor.chip(line,1) = axis_input;
204 }
205
206
207 //##########################################################################
208 // second phase
209 //##########################################################################
210
211 //--------------------------------------------------------------------------
212 // distribution
213 //--------------------------------------------------------------------------
214
215 // Create the block distribution in the second phase of the transformation
216 // transposing block_distribution
217 const Eigen::MatrixXi block_distribution_2phase = block_distribution.transpose();
218
219 /*
220 * Preparing MPI_Alltoallv data structures
221 */
222
223 // sending
224 std::vector<int> send_counts(size, x_block_size);
225 // Is an integer vector where entry i specifies the displacement (offset from sendbuf, in units of sendtype)
226 // from which to send data to rank i.
227 std::vector<int> send_displ(size);
228
229 for(int i=0; i<size; i++){
230 // get where to send each block thanks to the block_distribution_2phase
231 send_displ[i] = block_distribution_2phase(i,rank)*x_block_size;
232 }
233
234 // for all columns of the local_tensor we have to perform an MPI_Alltoallv due to col-major format
235 for(unsigned int local_col_index=0; local_col_index<y_local_dimension_size;local_col_index++){
236 int starting_index_of_the_local_col = local_col_index*x_dimension_size;
237 MPI_Alltoallv(MPI_IN_PLACE, NULL, NULL, MPI_DATATYPE_NULL,
238 &local_tensor.data()[starting_index_of_the_local_col], send_counts.data(), send_displ.data(), mpi_datatype, MPI_COMM_WORLD);
239 }
240
241 CTensor_2D local_tensor_transposed(x_block_size, y_block_size*size);
242 const unsigned int x_local_transposed_dimension_size = x_block_size;
243 //const unsigned int y_local_transposed_dimension_size = y_block_size*size;
244
245 // block transposition in order to use the
246 for(int n_block=0; n_block<size; n_block++)
247 {
248 for(unsigned int i=0; i<x_block_size; i++){
249 for(unsigned int j=0; j<y_block_size; j++){
250 local_tensor_transposed(i,j+n_block*y_block_size) = local_tensor(i+n_block*x_block_size,j);
251 }
252 }
253 }
254
255
256 //--------------------------------------------------------------------------
257 // transformation
258 //--------------------------------------------------------------------------
259 for(Eigen::Index line=0; line< x_local_transposed_dimension_size; line++){
260 axis_input = local_tensor_transposed.chip(line,0);
261 omp_1D_solver.fft(axis_input, fftDirection);
262 local_tensor_transposed.chip(line,0) = axis_input;
263 }
264
265
266 //##########################################################################
267 // final phase
268 //##########################################################################
269
270 //gather all rows
271 int col_local, col_global;
272 for(unsigned int j=0; j<y_dimension_size; j++){
273 col_local = j*x_block_size;
274 col_global = j*x_dimension_size;
275
276 MPI_Gather(&local_tensor_transposed.data()[col_local], x_block_size, mpi_datatype,
277 &global_tensor.data()[col_global], x_block_size, mpi_datatype, 0, MPI_COMM_WORLD );
278 }
279
280 }
281}
