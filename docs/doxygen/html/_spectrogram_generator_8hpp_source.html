87 void SpectrogramGenerator<FFTStrategy>::load_audio(CTensor_1D &signal, TensorIdx frame_lenght, TensorIdx frame_step, bool pad_end){
88
89 TensorIdx signal_lenght = signal.get_tensor().size();
90
91 //if frame_step is not specified, it is set to frame_lenght
92 if(frame_step == 0){
93 frame_step = frame_lenght;
94 }
95
96 assert(signal_lenght >= frame_lenght && "Signal lenght must be greater than frame lenght");
97 assert((frame_lenght & (frame_lenght - 1)) == 0 && "Frame lenght must be a power of 2");
98 assert(frame_lenght >= frame_step && "Frame lenght must be greater than frame step");
99
100 //compute the number of frames(excluding the last one if pad_end is false)
101 TensorIdx num_frames = (signal_lenght - frame_lenght) / frame_step + 1;
102
103 //clear the data vector and reserve space for the frames
104 _transform_buffer.clear();
105 _transform_buffer.reserve(num_frames);
106
107 //load the frames into the data vector
108 for(TensorIdx i = 0; i < num_frames; i++){
109 _transform_buffer.push_back(Frame_t(signal, frame_lenght, frame_step, i));
110 }
111
112 //if pad_end is true, the last frame is loaded separately
113 TensorIdx last_frame_start = num_frames * frame_step;
114 if(pad_end && last_frame_start < signal_lenght){
115 _transform_buffer.push_back(Frame_t(signal, frame_lenght, last_frame_start));
116 num_frames++;
117 }
118
119 //store the current parameters
120 _signal_lenght = signal_lenght;
121 _frame_lenght = frame_lenght;
122 _num_frames = num_frames;
123
124 _is_loaded = true;
125 }

154 void SpectrogramGenerator<FFTStrategy>::compute_spectrogram(bool decibel_scale, bool assume_real){
155
156 //check that a signal has been loaded
157 assert(_is_loaded && "No signal loaded");
158
159 //for slicing with Eigen
160 using Index1D = std::array<TensorIdx, 1>;
161
162 //compute the FFT of each frame
163 for(auto &frame : _transform_buffer){
164 _fft_solver.compute_fft(frame, FFT_FORWARD);
165 }
166
167 //compute the number of bins in the real and complex case
168 unsigned int num_bins = assume_real ? _frame_lenght / 2 + 1 : _frame_lenght;
169
170 //construct the spectrogram
171 Spectrogram_t spectrogram(num_bins, _num_frames);
172
173 //copy the FFT results into the spectrogram and square the absolute value to get the power spectrum
174 Index1D offset = {0};
175 Index1D extent = {num_bins};
176 for(TensorIdx i = 0; i < _num_frames; i++){
177 spectrogram.get_tensor().chip(i, 1) = _transform_buffer[i].get_tensor().slice(offset, extent).abs().square();
178 }
179
180 //convert to decibel scale if required
181 if(decibel_scale){
182 FloatingType epsilon = std::numeric_limits<FloatingType>::epsilon();
183 FloatingType scale_factor = 10;
184 auto power_to_decibel = [epsilon, scale_factor](FloatingType x){ return scale_factor * std::log10(x + epsilon); };
185 spectrogram.get_tensor() = spectrogram.get_tensor().unaryExpr(power_to_decibel);
186 }
187
188 //spectrogram is moved into the vector
189 _spectrograms.push_back(std::move(spectrogram));
190 }
