52 template <typename FloatingType>
53 void StockhamFFT<FloatingType>::fft(CTensor_1D &input_output, FFTDirection fftDirection) const
54 {
55 using Complex = std::complex<FloatingType>;
56
57 CTensor_1D &input = input_output;
58 CTensor_1D buffer(input.size());
59
60 const TensorIdx n = input.size();
61 assert(!(n & (n - 1)) && "FFT length must be a power of 2.");
62 const TensorIdx n2 = n >> 1; // n2 = n/2
63
64 const unsigned int log2n = std::log2(n);
65
66 //conjugate if inverse
67 if(fftDirection == fftcore::FFT_INVERSE){
68 FFTUtils::conjugate(input);
69 }
70
71 Complex w, t, u;
72 TensorIdx m, m2, r, r2;
73 // Stockham iterative FFT
74 for(unsigned int s = 1; s <= log2n; ++s){
75
76 //std::cout << "s = " << s << std::endl;
77
78 m = 1 << s; // 2^s
79 m2 = m >> 1; // m2 = m/2
80 r = n / m; // r = n/m
81 r2 = r << 1; // r2 = 2r
82
83 for(TensorIdx j = 0; j < m2; ++j){
84
85 w = exp(Complex(0, -2 * M_PI * j / m)); // w = e^(-2*pi*j/m)
86
87 for(TensorIdx k = 0; k < r; ++k){
88
89 u = input[j * r2 + k];
90 t = w * input[j * r2 + k + r];
91
92 buffer[j * r + k] = u + t;
93 buffer[j * r + n2 + k] = u - t;
94 }
95 }
96 std::swap(input, buffer);
97 }
98
99 //re-conjugate and scale if inverse
100 if(fftDirection == fftcore::FFT_INVERSE){
101 FFTUtils::conjugate(input);
102 input = input * Complex(1.0/n, 0);
103 }
104 };

106}
