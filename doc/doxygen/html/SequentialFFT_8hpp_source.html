52 template <typename FloatingType>
53 void SequentialFFT<FloatingType>::fft(CTensor_1D &input_output, FFTDirection fftDirection) const
54 {
55
56 using Complex = std::complex<FloatingType>;
57
58 const TensorIdx n = input_output.size();
59 assert(!(n & (n - 1)) && "FFT length must be a power of 2.");
60
61 const unsigned int log2n = std::log2(n);
62
63 //conjugate if inverse
64 if(fftDirection == FFT_INVERSE){
65 FFTUtils::conjugate(input_output);
66 }
67
68 // Bit-reversal permutation
69 FFTUtils::bit_reversal_permutation(input_output);
70
71 Complex w, wm, t, u;
72 TensorIdx m, m2;
73 // Cooley-Tukey iterative FFT
74 for (unsigned int s = 1; s <= log2n; ++s)
75 {
76 m = 1 << s; // 2 power s
77 m2 = m >> 1; // m2 = m/2 -1
78 wm = exp(Complex(0, -2 * M_PI / m)); // w_m = e^(-2*pi/m)
79
80 for(TensorIdx k = 0; k < n; k += m)
81 {
82 w = Complex(1, 0);
83 for(TensorIdx j = 0; j < m2; ++j)
84 {
85 t = w * input_output[k + j + m2];
86 u = input_output[k + j];
87
88 input_output[k + j] = u + t;
89 input_output[k + j + m2] = u - t;
90
91 w *= wm;
92 }
93 }
94 }
95
96 //re-conjugate and scale if inverse
97 if(fftDirection == FFT_INVERSE){
98 FFTUtils::conjugate(input_output);
99 input_output = input_output * Complex(1.0/n, 0);
100 }
101 };
