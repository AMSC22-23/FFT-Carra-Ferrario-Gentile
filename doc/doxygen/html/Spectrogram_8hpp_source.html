64 void SpectrogramGenerator<FFTStrategy>::load_audio(CTensor_1D &signal, unsigned int frame_lenght, unsigned int frame_step, bool pad_end){
65
66 unsigned int signal_lenght = signal.get_tensor().size();
67
68 //if frame_step is not specified, it is set to frame_lenght
69 if(frame_step == 0){
70 frame_step = frame_lenght;
71 }
72
73 assert(signal_lenght >= frame_lenght && "Signal lenght must be greater than frame lenght");
74 assert((frame_lenght & (frame_lenght - 1)) == 0 && "Frame lenght must be a power of 2");
75 assert(frame_lenght >= frame_step && "Frame lenght must be greater than frame step");
76
77 //compute the number of frames(excluding the last one if pad_end is false)
78 unsigned int num_frames = (signal_lenght - frame_lenght) / frame_step + 1;
79
80 _data.clear();
81 _data.reserve(num_frames);
82
83 for(unsigned int i = 0; i < num_frames; i++){
84 _data.push_back(Frame_t(signal, frame_lenght, frame_step, i));
85 }
86
87 unsigned int last_frame_start = num_frames * frame_step;
88
89 if(pad_end && last_frame_start < signal_lenght){
90 _data.push_back(Frame_t(signal, frame_lenght, last_frame_start));
91 num_frames++;
92 }
93
94 _signal_lenght = signal_lenght;
95 _frame_lenght = frame_lenght;
96 _num_frames = num_frames;
97 }

104 template<class FFTStrategy>
105 void SpectrogramGenerator<FFTStrategy>::compute_spectrogram(bool assume_real){
106
107 //for slicing with Eigen
108 using Index1D = std::array<Eigen::Index, 1>;
109
110 //compute the FFT of each frame
111 for(auto &frame : _data){
112 _fft_solver.compute_fft(frame, FFT_FORWARD);
113 }
114
115 //compute the number of bins in the real and complex case
116 unsigned int num_bins = assume_real ? _frame_lenght / 2 + 1 : _frame_lenght;
117
118 //construct the spectrogram
119 Spectrogram_t spectrogram(num_bins, _num_frames);
120
121 //copy the FFT results into the spectrogram
122 Index1D offset = {0};
123 Index1D extent = {num_bins};
124 for(unsigned int i = 0; i < _num_frames; i++){
125 //copy the first half of each fft result and take the absolute value
126 spectrogram.get_tensor().chip(i, 1) = _data[i].get_tensor().slice(offset, extent).abs();
127 }
128
129 //spectrogram is moved into the vector
130 _spectrograms.push_back(std::move(spectrogram));
131 }
